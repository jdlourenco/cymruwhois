def whois(addr)
  ip = IPAddr.new(addr)
  if ip.ipv4?
    revdns = ip.reverse.sub("in-addr.arpa", ORIGIN)
  elsif ip.ipv6?
    revdns = ip.reverse.sub("ip6.arpa", ORIGIN6)
  end
  
  ansip = intxt(revdns)
  @asnum = ansip[0]
  @cidr = ansip[1]
  @country = ansip[2]
  @registry = ansip[3]
  @allocdate = ansip[4]
  
  ansasnum = Cymru::ASNumber.new
  ansasnum.whois(@asnum)
  @asname = ansasnum.asname
  
  ansip << @asname
end
